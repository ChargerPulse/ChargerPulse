import { createClient } from '@supabase/supabase-js'

const supabaseUrl = process.env.NEXT_PUBLIC_SUPABASE_URL!
const supabaseKey = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
const supabase = createClient(supabaseUrl, supabaseKey)

export async function GET() {
  try {
    // Fetch all chargers
    const { data: chargers, error: chargersError } = await supabase
      .from('chargers')
      .select('*')

    if (chargersError) throw chargersError

    // For each charger, calculate uptime
    const chargerData = await Promise.all(
      chargers.map(async (charger) => {
        // Get events for the last 30 days
        const thirtyDaysAgo = new Date()
        thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30)

        const { data: events } = await supabase
          .from('events')
          .select('status, ts')
          .eq('cp_id', charger.id)
          .gte('ts', thirtyDaysAgo.toISOString())

        if (!events || events.length === 0) {
          return {
            id: charger.id,
            uptime24h: 0,
            uptime7d: 0,
            uptime30d: 0,
            lastUpdate: new Date().toLocaleTimeString()
          }
        }

        // Calculate uptime percentages
        const now = new Date()
        const oneDayAgo = new Date(now.getTime() - 24 * 60 * 60 * 1000)
        const sevenDaysAgo = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000)

        const available24h = events.filter(
          (e) => new Date(e.ts) > oneDayAgo && e.status === 'Available'
        ).length
        const total24h = events.filter(
          (e) => new Date(e.ts) > oneDayAgo
        ).length

        const available7d = events.filter(
          (e) => new Date(e.ts) > sevenDaysAgo && e.status === 'Available'
        ).length
        const total7d = events.filter(
          (e) => new Date(e.ts) > sevenDaysAgo
        ).length

        const available30d = events.filter(
          (e) => e.status === 'Available'
        ).length
        const total30d = events.length

        return {
          id: charger.id,
          uptime24h: total24h > 0 ? ((available24h / total24h) * 100).toFixed(1) : 0,
          uptime7d: total7d > 0 ? ((available7d / total7d) * 100).toFixed(1) : 0,
          uptime30d: total30d > 0 ? ((available30d / total30d) * 100).toFixed(1) : 0,
          lastUpdate: events[events.length - 1]?.ts
            ? new Date(events[events.length - 1].ts).toLocaleTimeString()
            : 'N/A'
        }
      })
    )

    return Response.json(chargerData)
  } catch (error) {
    console.error('Error fetching chargers:', error)
    return Response.json({ error: 'Failed to fetch chargers' }, { status: 500 })
  }
}
